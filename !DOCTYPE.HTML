Arduino Control Code

// Motor control pins
int IN1 = 8;
int IN2 = 9;

void setup() {

  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);

}

void loop() {

  // =========================
  // CLOSE FINGER
  // =========================

  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);

  delay(2000);

  // STOP MOTOR
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);

  delay(1000);

  // =========================
  // OPEN FINGER
  // =========================

  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);

  delay(2000);

  // STOP MOTOR
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);

  delay(1000);
}
